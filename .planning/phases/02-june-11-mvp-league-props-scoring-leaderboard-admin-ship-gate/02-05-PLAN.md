<objective>
Ship the admin score-entry surface: an unlocalized RSC at `/admin/matches` that mirrors the player matches feed (chronological, date-grouped) but with a URL-driven `[View Mode] | [Score Entry Mode]` toggle. In Entry Mode each row exposes two number inputs + a Save Result button. Clicking Save fires `saveResult` — a service-role Server Action that updates `fixtures.result_*_90min`, reads all `predictions` for that fixture, scores each via `scoreMatch` (Plan 02-02), and calls `sweepAndUpsert` to idempotently push `score_events` rows + revalidate the 8 explicit per-locale paths.

**Wave/dependency note:** This plan's *logical* prerequisites are Plans 01 (schema), 02 (scoring helpers), and 03 (player matches feed it mirrors). Plan 04 is listed in `depends_on` purely as a serialization gate — both 02-04 and 02-05 mutate `messages/en.json`, so 02-05 MUST run after 02-04's commit lands to avoid an `Edit`-tool stale-state conflict on the shared JSON file. The wave stays at 4 (one wave after 02-04 at wave 3) per the file-modify rule in `<assign_waves>`.

Purpose: This is the second-half of the loop: predict (Plan 02-03) → admin enters result (HERE) → leaderboard updates (Plan 02-07). Without admin score entry, scores are forever 0. ADM-01, ADM-02, SCR-01, SCR-02, SCR-06, LB-03 all close here.

Output: Admin can enter a result for any fixture and the leaderboard recomputes — idempotently, no double-counting, no manual refresh.

    ```typescript
    // src/app/actions/saveResult.ts
    'use server';

    import { requireAdmin } from '@/lib/auth/session';
    import { createServiceClient } from '@/lib/supabase/service';
    import { resultSchema } from '@/lib/schemas/result';
    import { scoreMatch, type LeagueKind } from '@/lib/scoring/league';
    import { sweepAndUpsert } from '@/lib/scoring/sweepAndUpsert';
    import type { Database } from '@/types/supabase';

    type ScoreEventInsert = {
      user_id: string;
      source: 'league';
      ref_id: string;
      points: number;
      kind: LeagueKind;
    };

    export type SaveResultResponse =
      | { ok: true; scored: number }
      | { ok: false; error: string };

    /**
     * D-10 + D-18: overwrite-to-correct + idempotent UPSERT sweep.
     *
     * 1. requireAdmin() — gates the entire flow; redirects non-admin to /admin/403.
     * 2. Zod validate (resultSchema: fixture_id uuid, scores 0-9).
     * 3. UPDATE fixtures.result_home_90min / result_away_90min (D-12 — _full stays null in Phase 2).
     * 4. SELECT all predictions for this fixture (service-role bypasses RLS — required so the
     *    admin sees every member's pick, not just their own per Pitfall 10).
     * 5. Score each in TS via scoreMatch (Plan 02-02).
     * 6. sweepAndUpsert (DELETE stragglers + bulk UPSERT on PK + revalidatePath x8).
     */
    export async function saveResult(input: unknown): Promise<SaveResultResponse> {
      await requireAdmin();

      const parsed = resultSchema.safeParse(input);
      if (!parsed.success) return { ok: false, error: 'validation' };
      const { fixture_id, result_home_90min, result_away_90min } = parsed.data;

      const svc = createServiceClient();

      // 1. Persist the result. D-12: only _90min in Phase 2.
      const { error: e1 } = await svc
        .from('fixtures')
        .update({
          result_home_90min,
          result_away_90min,
          // updated_at: column may or may not exist depending on Phase 1 shape; OMIT to be safe.
          // If a Phase 1 trigger exists to bump updated_at on UPDATE, it fires automatically.
        })
        .eq('id', fixture_id);
      if (e1) return { ok: false, error: `fixture_update:${e1.message}` };

      // 2. Read all predictions for this fixture (service-role bypasses RLS).
      const { data: preds, error: e2 } = await svc
        .from('predictions')
        .select('user_id, home_score, away_score')
        .eq('fixture_id', fixture_id);
      if (e2) return { ok: false, error: `predictions_read:${e2.message}` };

      // 3. Score each.
      const rows: ScoreEventInsert[] = (preds ?? []).map((p) => {
        const { points, kind } = scoreMatch(
          { home_score: p.home_score, away_score: p.away_score },
          { result_home_90min, result_away_90min },
        );
        return {
          user_id: p.user_id,
          source: 'league' as const,
          ref_id: fixture_id,
          points,
          kind,
        };
      });

      // 4. Sweep + UPSERT + revalidate (shared helper from Plan 02-02).
      const sweepResult = await sweepAndUpsert({
        svc,
        source: 'league',
        ref_id: fixture_id,
        rows,
      });

      if (!sweepResult.ok) return { ok: false, error: sweepResult.error };

      return { ok: true, scored: rows.length };
    }
    ```

    ```typescript
    'use client';

    import Link from 'next/link';
    import { useSearchParams } from 'next/navigation';
    import { useTranslations } from 'next-intl';

    /**
     * D-09 + UI-SPEC §11: segmented control bound to URLSearchParam ?mode=view|entry.
     * Default view. Sticky? No — let it scroll away with the page.
     *
     * Uses next/navigation's useSearchParams + a literal href update (not Server Action)
     * because /admin/* is unlocalized and we don't need to round-trip through the server.
     */
    export function AdminModeToggle() {
      const t = useTranslations('admin.modeToggle');
      const sp = useSearchParams();
      const mode = sp.get('mode') === 'entry' ? 'entry' : 'view';

      // Build target hrefs preserving any other search params (none expected on /admin/matches in Phase 2).
      const next = (target: 'view' | 'entry') => {
        const params = new URLSearchParams(sp.toString());
        params.set('mode', target);
        return `/admin/matches?${params.toString()}`;
      };

      const baseSeg = 'flex-1 bs-10 ps-4 pe-4 inline-flex items-center justify-center rounded-lg text-base font-bold transition-colors duration-150';
      const active = 'bg-[var(--zc-card)] text-[var(--zc-primary)] shadow-sm';
      const inactive = 'bg-transparent text-[var(--zc-muted-foreground)]';

      return (
        <nav className="inline-flex bs-12 rounded-xl bg-[var(--zc-muted)] p-1" aria-label="Mode">
          <Link
            href={next('view')}
            replace
            scroll={false}
            className={`${baseSeg} ${mode === 'view' ? active : inactive}`}
            aria-current={mode === 'view' ? 'page' : undefined}
          >
            {t('view')}
          </Link>
          <Link
            href={next('entry')}
            replace
            scroll={false}
            className={`${baseSeg} ${mode === 'entry' ? `${active} border-s-2 border-[var(--zc-accent)]` : inactive}`}
    ```

    ```typescript
    // src/components/admin/AdminResultInputs.client.tsx
    'use client';

    import { useState, useTransition } from 'react';
    import { useTranslations } from 'next-intl';
    import { saveResult } from '@/app/actions/saveResult';

    type Props = {
      fixtureId: string;
      initialHome: number | null;
      initialAway: number | null;
    };

    export function AdminResultInputs({ fixtureId, initialHome, initialAway }: Props) {
      const t = useTranslations('admin.saveResult');
      const [home, setHome] = useState<string>(initialHome !== null ? String(initialHome) : '');
      const [away, setAway] = useState<string>(initialAway !== null ? String(initialAway) : '');
      const [state, setState] = useState<'idle' | 'saving' | 'saved' | 'failed'>('idle');
      const [errorMsg, setErrorMsg] = useState<string | null>(null);
      const [isPending, startTransition] = useTransition();

      const handleSave = () => {
        setState('saving');
        setErrorMsg(null);
        startTransition(async () => {
          const result = await saveResult({
            fixture_id: fixtureId,
            result_home_90min: parseInt(home, 10),
            result_away_90min: parseInt(away, 10),
          });
          if (result.ok) {
            setState('saved');
            setTimeout(() => setState('idle'), 2000);
          } else {
            setState('failed');
            setErrorMsg(result.error);
          }
        });
      };

      const buttonLabel =
        state === 'saving' ? t('saving') :
        state === 'saved'  ? t('saved')  :
                              t('idle');

      const inputClass = 'bs-12 is-12 text-center text-2xl font-bold rounded-xl border border-[var(--zc-border)] focus-visible:outline-none focus-visible:border-[var(--zc-accent)] focus-visible:ring-2 focus-visible:ring-[var(--zc-ring)] tabular-nums';

      return (
        <div className="inline-flex items-center gap-2">
          <input
            type="number"
            inputMode="numeric"
            min={0}
            max={9}
            dir="ltr"
            value={home}
            onChange={(e) => setHome(e.target.value)}
            className={inputClass}
            aria-label="home score result"
          />
          <span className="text-2xl text-[var(--zc-muted-foreground)]">:</span>
          <input
            type="number"
            inputMode="numeric"
            min={0}
            max={9}
            dir="ltr"
            value={away}
            onChange={(e) => setAway(e.target.value)}
            className={inputClass}
            aria-label="away score result"
          />
          <button
            type="button"
            onClick={handleSave}
            disabled={isPending || home === '' || away === ''}
            className="bs-12 ps-4 pe-4 rounded-xl bg-[var(--zc-primary)] text-[var(--zc-primary-foreground)] font-bold hover:bg-[#13325a] focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-[var(--zc-ring)] focus-visible:ring-offset-2 focus-visible:ring-offset-[var(--zc-card)] disabled:opacity-60"
          >
            {buttonLabel}
          </button>
          {state === 'failed' && (
            <span className="ms-2 text-sm text-[var(--zc-destructive)]" role="alert">
              {t('failed')}{errorMsg ? ` (${errorMsg})` : ''}
            </span>
          )}
        </div>
      );
    }
    ```

    ```typescript
    import { adminReadClient } from '@/lib/auth/adminReadClient';
    import { groupByLocalDate } from '@/lib/matches/groupByLocalDate';
    import { DateGroupHeader } from '@/components/matches/DateGroupHeader';
    import { MatchRowLocked } from '@/components/matches/MatchRowLocked';
    import { MatchRowResulted } from '@/components/matches/MatchRowResulted';
    import { AdminModeToggle } from '@/components/admin/AdminModeToggle.client';
    import { AdminResultInputs } from '@/components/admin/AdminResultInputs.client';
    import type { PtsKind } from '@/components/ui/PtsBadge';

    type Props = { searchParams: Promise<{ mode?: 'view' | 'entry' }> };

    export default async function AdminMatchesPage({ searchParams }: Props) {
      const sp = await searchParams;
      const mode = sp.mode === 'entry' ? 'entry' : 'view';

      // requireAdmin() is already enforced by /admin/(protected)/layout.tsx (Phase 1).
      // adminReadClient() composes requireAdmin + service-role to bypass RLS (Pitfall 10).
      const svc = await adminReadClient();

      // Fetch fixtures + nested team FKs + predictions + score_events in two queries.
      const { data: fixtures } = await svc
        .from('fixtures')
        .select(`
          id, external_match_no, stage, group_code, kickoff_at,
          home_placeholder, away_placeholder,
          result_home_90min, result_away_90min,
          home_team:teams!fixtures_home_team_id_fkey ( id, code, name_en, name_he ),
          away_team:teams!fixtures_away_team_id_fkey ( id, code, name_en, name_he ),
          predictions ( user_id, home_score, away_score, submitted_at,
                        profiles:user_id ( display_name ) )
        `)
        .order('kickoff_at', { ascending: true });

      if (!fixtures || fixtures.length === 0) {
        return (
          <main className="pi-4 pbs-4">
            <p className="text-base text-[var(--zc-muted-foreground)]">No matches yet — check the seed migration.</p>
          </main>
        );
      }

      const fixtureIds = fixtures.map((f) => f.id);
      const { data: scoreEventsData } = await svc
        .from('score_events')
        .select('user_id, ref_id, points, kind')
        .eq('source', 'league')
        .in('ref_id', fixtureIds);
      const scoreByKey = new Map<string, { points: number; kind: PtsKind | null }>();
      for (const se of scoreEventsData ?? []) {
        scoreByKey.set(`${se.ref_id}:${se.user_id}`, { points: se.points, kind: (se.kind as PtsKind | null) });
      }

      const { data: roster } = await svc.from('profiles').select('user_id, display_name');

      const now = Date.now();
      const groups = groupByLocalDate(fixtures, 'en');

      return (
        <main className="pi-4 pbs-4 pbe-24">
          <div className="mbe-4">
            <AdminModeToggle />
          </div>
          {groups.map(([dateLabel, fxs]) => (
            <section key={dateLabel}>
              <DateGroupHeader label={dateLabel} />
              {fxs.map((f) => {
                const kickoffMs = new Date(f.kickoff_at).getTime();
                const isLocked = kickoffMs <= now;
                const isResulted = f.result_home_90min !== null && f.result_away_90min !== null;
                const homeTeam = f.home_team ?? { id: '?', code: '?', name_en: f.home_placeholder ?? '?', name_he: f.home_placeholder ?? '?' };
                const awayTeam = f.away_team ?? { id: '?', code: '?', name_en: f.away_placeholder ?? '?', name_he: f.away_placeholder ?? '?' };

                if (mode === 'entry') {
                  // Entry mode: show input + Save button on EVERY row (corrections allowed).
                  return (
                    <div key={f.id} className="bg-[var(--zc-card)] border border-[var(--zc-border)] rounded-2xl pi-4 pbs-3 pbe-3 mbs-3 min-bs-16 flex items-center gap-4">
                      <div className="flex items-center gap-2 min-is-0 flex-1">
                        <span className="text-xl" aria-hidden>🏴</span>
                        <span className="text-base truncate">{homeTeam.name_en}</span>
                      </div>
                      <AdminResultInputs
                        fixtureId={f.id}
                        initialHome={f.result_home_90min}
                        initialAway={f.result_away_90min}
                      />
                      <div className="flex items-center gap-2 min-is-0 flex-1 justify-end">
                        <span className="text-base truncate text-end">{awayTeam.name_en}</span>
                        <span className="text-xl" aria-hidden>🏴</span>
                      </div>
                    </div>
                  );
                }

                // View mode: same variants as player /matches.
                if (isResulted) {
                  const picks = (roster ?? []).map((profile) => {
                    const pred = (f.predictions ?? []).find((p) => p.user_id === profile.user_id);
                    const score = scoreByKey.get(`${f.id}:${profile.user_id}`);
                    return {
                      user_id: profile.user_id,
                      display_name: profile.display_name,
                      home_score: pred?.home_score ?? null,
                      away_score: pred?.away_score ?? null,
                      points: score?.points ?? 0,
                      kind: (score?.kind ?? null) as PtsKind | null,
                    };
                  });
                  return (
                    <MatchRowResulted
                      key={f.id}
                      locale="en"
                      homeTeam={homeTeam}
                      awayTeam={awayTeam}
                      kickoffAt={f.kickoff_at}
                      resultHome={f.result_home_90min!}
                      resultAway={f.result_away_90min!}
                      picks={picks}
                    />
                  );
                }

                if (isLocked) {
                  return (
                    <MatchRowLocked
                      key={f.id}
                      locale="en"
                      homeTeam={homeTeam}
                      awayTeam={awayTeam}
                      kickoffAt={f.kickoff_at}
                      userHome={null}  // admin doesn't have personal picks in admin view
                      userAway={null}
                    />
                  );
                }

                // Pre-kickoff, View Mode, admin sees the editable shape replaced by read-only summary.
                return (
                  <div key={f.id} className="bg-[var(--zc-card)] border border-[var(--zc-border)] rounded-2xl pi-4 pbs-3 pbe-3 mbs-3 min-bs-16 flex items-center gap-4">
                    <div className="flex items-center gap-2 min-is-0 flex-1">
                      <span className="text-xl" aria-hidden>🏴</span>
                      <span className="text-base truncate">{homeTeam.name_en}</span>
                    </div>
                    <span className="text-sm text-[var(--zc-muted-foreground)]">Not yet kicked off</span>
                    <div className="flex items-center gap-2 min-is-0 flex-1 justify-end">
                      <span className="text-base truncate text-end">{awayTeam.name_en}</span>
                      <span className="text-xl" aria-hidden>🏴</span>
                    </div>
                  </div>
                );
              })}
            </section>
          ))}
        </main>
      );
    }
    ```
