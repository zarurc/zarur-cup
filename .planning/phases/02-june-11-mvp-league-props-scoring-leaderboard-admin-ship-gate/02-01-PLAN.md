    ```sql
    -- Migration 0007: score_events — idempotent points projection (D-19)
    -- Phase 2 saveResult / gradeProp Server Actions UPSERT into this table;
    -- v_leaderboard (0008) aggregates. PK (user_id, source, ref_id) IS the
    -- idempotency key per SCR-06 + D-10. RLS: SELECT for all members (per
    -- D-19 / family-trust); no INSERT/UPDATE/DELETE for authenticated (service
    -- role bypasses for writes per RESEARCH Pattern 5).

    create table public.score_events (
      user_id     uuid not null references auth.users(id) on delete cascade,
      source      text not null check (source in ('league','prop','bracket')),
      ref_id      uuid not null,
      points      smallint not null,
      kind        text     null check (kind in ('exact','goal-diff','winner','miss','correct')),
      updated_at  timestamptz not null default now(),
      primary key (user_id, source, ref_id)
    );

    create index score_events_user_idx on public.score_events (user_id);
    create index score_events_source_ref_idx on public.score_events (source, ref_id);

    alter table public.score_events enable row level security;

    -- SELECT for all signed-in members (D-19, RESEARCH lines 875-879).
    create policy score_events_read on public.score_events
      for select to authenticated using (true);

    -- No INSERT/UPDATE/DELETE policies => denied by default. Service-role writes only.

    -- GRANTs (mirror 0003_grants.sql + 0004_anon_select.sql shape).
    grant select on public.score_events to authenticated;
    grant select on public.score_events to anon;
    grant select, insert, update, delete on public.score_events to service_role;

    -- B-style smoke: confirm authenticated has NO DML grants (mirror 0002 lines 223-237).
    do $$
    declare cnt int;
    begin
      select count(*) into cnt
        from information_schema.column_privileges
        where table_schema = 'public'
          and table_name = 'score_events'
          and grantee = 'authenticated'
          and privilege_type in ('INSERT','UPDATE','DELETE');

        raise exception 'score_events B-style check failed: authenticated has unexpected DML grant';
      end if;
    end$$;
    ```

    **0008_v_leaderboard.sql — per D-20 + RESEARCH lines 615-643:**

    ```sql
    -- Migration 0008: v_leaderboard view — aggregates score_events per profile.
    -- Tiebreaker ORDER BY lives in the consumer RSC (TS-side Intl.Collator per
    -- RESEARCH Pattern 7 / Pitfall 5 — avoids ICU availability question).
    -- Bracket subtotal stays 0 until Phase 3 writes source='bracket' rows.

    create or replace view public.v_leaderboard as
    select
      p.user_id,
      p.display_name,
      coalesce(sum(se.points), 0)::int                                                              as total,
      coalesce(sum(se.points) filter (where se.source = 'league'),  0)::int                          as league_total,
      coalesce(sum(se.points) filter (where se.source = 'prop'),    0)::int                          as props_total,
      coalesce(sum(se.points) filter (where se.source = 'bracket'), 0)::int                          as bracket_total,
      coalesce(count(*) filter (where se.kind = 'exact'), 0)::int                                    as exact_count,
      coalesce(count(*) filter (where se.kind in ('exact','goal-diff','winner','correct')), 0)::int  as correct_count
    from public.profiles p
    left join public.score_events se on se.user_id = p.user_id
    group by p.user_id, p.display_name;

    grant select on public.v_leaderboard to authenticated;
    grant select on public.v_leaderboard to anon;

    -- Smoke: view exists and is queryable by service_role.
    do $$
    begin
      perform 1 from public.v_leaderboard limit 1;
    exception when others then
      raise exception 'v_leaderboard smoke failed: %', sqlerrm;
    end$$;
    ```
